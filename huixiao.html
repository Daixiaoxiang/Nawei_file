就业率魔术公式


# 就业率计算公式（行业通用）
def employment_rate(signed_students):
    employed = 0
    for s in signed_students:
        if s.received_offer >= 1:  # 收到Offer就算就业
            employed += 1
        elif s.rejected_offers >= 3: # 拒接3次也算就业
            employed += 1 
    return employed / len(signed_students) * 100  # 公布：91%就业率
